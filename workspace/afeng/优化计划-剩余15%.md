### 任务1：表格列右对齐（优先级：高）

#### 问题分析
- **现象**：数字列有1-2px左右偏移
- **原因**：PyMuPDF提取的origin是左对齐位置，而原PDF的数字列是右对齐
- **影响**：表格数字列看起来不整齐

#### 解决方案
```python
# 在pdf_rebuilder.py的_draw_text方法中添加

def _detect_number_column(self, element, all_elements):
    """检测是否是数字列"""
    content = element["content"].strip()
    
    # 1. 检测内容是否是数字
    is_number = bool(re.match(r'^[\d.,\-+]+$', content))
    
    # 2. 检测是否在表格区域
    bbox = element["bbox"]
    x, y = bbox[0], bbox[1]
    
    # 查找同一列的其他元素
    same_column = [e for e in all_elements 
                   if e["type"] == "text" 
                   and abs(e["bbox"][0] - x) < 5  # 同一列
                   and e != element]
    
    # 3. 如果同列大部分是数字，则认为是数字列
    if same_column:
        number_count = sum(1 for e in same_column 
                          if re.match(r'^[\d.,\-+]+$', e["content"].strip()))
        if number_count / len(same_column) > 0.5:
            return True
    
    return is_number

def _draw_text_with_alignment(self, c, element, page_height, all_elements):
    """绘制文本，支持右对齐"""
    origin = element.get("origin")
    content = element["content"]
    
    if origin:
        x = origin[0]
        y = page_height - origin[1]
    else:
        x0, y0, x1, y1 = element["bbox"]
        x = x0
        y = page_height - y1
    
    # 设置字体
    font_name = self._get_font_name(element)
    font_size = element["font_size"]
    c.setFont(font_name, font_size)
    
    # 检测是否需要右对齐
    if self._detect_number_column(element, all_elements):
        # 计算文本宽度
        text_width = c.stringWidth(content, font_name, font_size)
        
        # 查找列的右边界（通过bbox的x1）
        bbox = element["bbox"]
        column_right = bbox[2]
        
        # 右对齐：从右边界往左偏移文本宽度
        x = column_right - text_width
    
    # 绘制文本
    c.drawString(x, y, content)
```

### 任务2：图表颜色优化（优先级：中）

#### 问题分析
- **现象**：柱状图背景略灰，饼图颜色略有差异
- **原因**：图像提取时的色彩空间转换
- **影响**：视觉上有轻微色差

#### 解决方案
```python
# 在pdf_rebuilder.py的_draw_image方法中优化

def _draw_image_optimized(self, c, element, page_height):
    """绘制图像，优化色彩空间"""
    from io import BytesIO
    from PIL import Image
    
    x0, y0, x1, y1 = element["bbox"]
    x = x0
    y = page_height - y1
    width = x1 - x0
    height = y1 - y0
    
    try:
        image_data = element["image_data"]
        image_stream = BytesIO(image_data)
        pil_img = Image.open(image_stream)
        
        # 1. 确保RGB模式
        if pil_img.mode in ('RGBA', 'LA', 'P'):
            rgb_img = Image.new('RGB', pil_img.size, (255, 255, 255))
            if pil_img.mode == 'RGBA':
                rgb_img.paste(pil_img, mask=pil_img.split()[3])
            else:
                rgb_img.paste(pil_img)
            pil_img = rgb_img
        
        # 2. 色彩空间优化
        # 检查是否需要色彩校正
        if self._is_chart_image(element):
            # 对图表图像进行色彩增强
            from PIL import ImageEnhance
            
            # 增强对比度（去除灰度）
            enhancer = ImageEnhance.Contrast(pil_img)
            pil_img = enhancer.enhance(1.1)  # 增强10%
            
            # 增强饱和度（颜色更鲜艳）
            enhancer = ImageEnhance.Color(pil_img)
            pil_img = enhancer.enhance(1.05)  # 增强5%
        
        # 3. 保存为高质量JPEG
        output_stream = BytesIO()
        pil_img.save(output_stream, format='JPEG', quality=98)  # 提高质量到98
        output_stream.seek(0)
        
        from reportlab.lib.utils import ImageReader
        img = ImageReader(output_stream)
        c.drawImage(img, x, y, width=width, height=height)
        
    except Exception as e:
        print(f"  警告: 无法绘制图像 @ {element['bbox']}: {e}")

def _is_chart_image(self, element):
    """检测是否是图表图像"""
    # 根据位置和大小判断
    bbox = element["bbox"]
    width = bbox[2] - bbox[0]
    height = bbox[3] - bbox[1]
    
    # 图表通常比较大（>100x100）且在页面下半部分
    is_large = width > 100 and height > 100
    is_lower_half = bbox[1] > 400  # 页面高度的一半
    
    return is_large and is_lower_half
```

### 任务3：头部间距微调（优先级：低）

#### 问题分析
- **现象**：Logo之间间距有2px偏差
- **原因**：图像位置的微小偏差
- **影响**：头部布局略有差异

#### 解决方案
```python
# 在pdf_rebuilder.py中添加头部区域特殊处理

def _adjust_header_spacing(self, elements):
    """调整头部区域的间距"""
    # 识别头部区域（顶部50px）
    header_images = [e for e in elements 
                     if e["type"] == "image" 
                     and e["bbox"][1] < 50]
    
    if len(header_images) < 2:
        return elements
    
    # 按X坐标排序
    header_images.sort(key=lambda e: e["bbox"][0])
    
    # 计算理想间距
    total_width = header_images[-1]["bbox"][2] - header_images[0]["bbox"][0]
    ideal_spacing = total_width / (len(header_images) - 1)
    
    # 调整位置
    for i, img in enumerate(header_images):
        if i == 0:
            continue  # 第一个不动
        
        # 计算新的X位置
        prev_img = header_images[i-1]
        new_x0 = prev_img["bbox"][2] + ideal_spacing
        
        # 调整bbox
        width = img["bbox"][2] - img["bbox"][0]
        img["bbox"] = (new_x0, img["bbox"][1], new_x0 + width, img["bbox"][3])
    
    return elements
```
